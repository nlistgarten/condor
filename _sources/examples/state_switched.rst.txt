        p2 = parameter()

        u = modal()

    class Accel(DblInt.Mode):
        condition = mode == 0.0
        action[u] = 1.0


    class Switch1(DblInt.Event):
        function = x[0] - p1
        update[mode] = 1.0


    class Decel(DblInt.Mode):
        condition = mode == 1.0
        action[u] = -1.0
